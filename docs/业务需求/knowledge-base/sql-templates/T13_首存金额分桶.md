# T13 首存金额分桶

## 模板用途

按首存金额固定档位输出人数与占比，并生成 Excel 同形的固定金额桶/占比交替列宽表。

## 建议问题（可转为 sql_pair.question）

- 生成首存金额分布与占比完整宽表，输出汇总和固定金额桶/占比交替列

## 核心表/模型

- dwd_order_deposit

## 参数

- tenant_plat_id
- channel_id
- start_date
- end_date

## SQL 模板

```sql
WITH first_deposit AS (
    SELECT
        DATE(d.callback_time) AS first_deposit_date,
        d.channel_id,
        d.player_id,
        d.actual_amount
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.times = 1
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :start_date
      AND d.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
),
bucketed AS (
    SELECT
        f.first_deposit_date,
        f.channel_id,
        f.player_id,
        CASE
            WHEN f.actual_amount = 10 THEN '10元'
            WHEN f.actual_amount = 20 THEN '20元'
            WHEN f.actual_amount = 30 THEN '30元'
            WHEN f.actual_amount = 50 THEN '50元'
            WHEN f.actual_amount = 100 THEN '100元'
            WHEN f.actual_amount = 200 THEN '200元'
            WHEN f.actual_amount = 300 THEN '300元'
            WHEN f.actual_amount = 400 THEN '400元'
            WHEN f.actual_amount = 500 THEN '500元'
            WHEN f.actual_amount = 1000 THEN '1000元'
            WHEN f.actual_amount = 2000 THEN '2000元'
            WHEN f.actual_amount > 2000 THEN '2000元以上'
            ELSE '其他金额'
        END AS bucket_name
    FROM first_deposit f
),
rollup_source AS (
    SELECT
        CAST(first_deposit_date AS CHAR) AS report_date,
        bucket_name,
        player_id
    FROM bucketed
    UNION ALL
    SELECT
        '汇总' AS report_date,
        bucket_name,
        player_id
    FROM bucketed
),
rollup_total AS (
    SELECT
        report_date,
        COUNT(DISTINCT player_id) AS first_deposit_user_count
    FROM rollup_source
    GROUP BY report_date
),
rollup_bucket AS (
    SELECT
        report_date,
        bucket_name,
        COUNT(DISTINCT player_id) AS bucket_user_count
    FROM rollup_source
    GROUP BY report_date, bucket_name
)
SELECT
    rt.report_date AS `日期`,
    rt.first_deposit_user_count AS `首存用户数`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '10元' THEN rb.bucket_user_count END), 0) AS `10元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '10元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `10元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '20元' THEN rb.bucket_user_count END), 0) AS `20元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '20元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `20元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '30元' THEN rb.bucket_user_count END), 0) AS `30元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '30元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `30元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '50元' THEN rb.bucket_user_count END), 0) AS `50元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '50元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `50元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '100元' THEN rb.bucket_user_count END), 0) AS `100元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '100元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `100元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '200元' THEN rb.bucket_user_count END), 0) AS `200元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '200元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `200元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '300元' THEN rb.bucket_user_count END), 0) AS `300元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '300元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `300元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '400元' THEN rb.bucket_user_count END), 0) AS `400元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '400元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `400元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '500元' THEN rb.bucket_user_count END), 0) AS `500元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '500元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `500元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '1000元' THEN rb.bucket_user_count END), 0) AS `1000元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '1000元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `1000元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '2000元' THEN rb.bucket_user_count END), 0) AS `2000元`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '2000元' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `2000元占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '2000元以上' THEN rb.bucket_user_count END), 0) AS `2000元以上`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '2000元以上' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `2000元以上占比`,
    COALESCE(SUM(CASE WHEN rb.bucket_name = '其他金额' THEN rb.bucket_user_count END), 0) AS `其他金额`,
    ROUND(COALESCE(SUM(CASE WHEN rb.bucket_name = '其他金额' THEN rb.bucket_user_count END), 0) / NULLIF(rt.first_deposit_user_count, 0), 6) AS `其他金额占比`
FROM rollup_total rt
LEFT JOIN rollup_bucket rb
       ON rb.report_date = rt.report_date
GROUP BY rt.report_date, rt.first_deposit_user_count
ORDER BY CASE WHEN rt.report_date = '汇总' THEN 0 ELSE 1 END, rt.report_date;
```

## 备注

- 使用 CASE WHEN 固定分桶，并输出汇总行与每日行，满足第一期 Excel FULL 同形校验。
- SQL 按 TiDB / MySQL 8 风格编写；导入前需在实际 runtime datasource 下做一次校验。
