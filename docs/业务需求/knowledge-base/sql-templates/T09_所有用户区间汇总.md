# T09 所有用户区间汇总

## 模板用途

统计某渠道在指定区间内全部用户/TOPN/非TOPN/VIP0~VIP3 的存款、充提差、有效投注、输赢、投充比、杀率。

## 建议问题（可转为 sql_pair.question）

- 统计某渠道在指定区间内全部用户的投充比和杀率
- 统计某渠道 TOP3 用户的投充比和杀率
- 统计某渠道在指定区间内全部用户/TOPN/非TOPN/VIP0~VIP3的存款、充提差、有效投注、输赢、投充比、杀率

## 核心表/模型

- dwd_order_deposit
- dwd_order_withdrawal
- dwd_bet_order

## 参数

- tenant_plat_id
- channel_id
- start_date
- end_date
- user_segment
- top_n

## SQL 模板

```sql
WITH deposit_activity_users AS (
    SELECT DISTINCT d.player_id
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :start_date
      AND d.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
),
withdraw_activity_users AS (
    SELECT DISTINCT w.player_id
    FROM dwd_order_withdrawal w
    WHERE w.status = 3
      AND w.tenant_plat_id = :tenant_plat_id
      AND w.channel_id = :channel_id
      AND w.callback_time >= :start_date
      AND w.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
),
bet_activity_users AS (
    SELECT DISTINCT b.player_id
    FROM dwd_bet_order b
    WHERE b.settle_status = 1
      AND b.tenant_plat_id = :tenant_plat_id
      AND b.channel_id = :channel_id
      AND b.settle_time >= :start_date
      AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
),
active_users AS (
    SELECT player_id FROM deposit_activity_users
    UNION
    SELECT player_id FROM withdraw_activity_users
    UNION
    SELECT player_id FROM bet_activity_users
),
active_user_dim AS (
    SELECT
        au.player_id,
        COALESCE(p.vip_id, 0) AS vip_id
    FROM active_users au
    LEFT JOIN dim_player p
           ON p.id = au.player_id
          AND p.tenant_plat_id = :tenant_plat_id
          AND p.channel_id = :channel_id
),
bet_rank_base AS (
    SELECT
        au.player_id,
        COALESCE(SUM(b.valid_bet_amount), 0) AS total_valid_bet_amount
    FROM active_users au
    LEFT JOIN dwd_bet_order b
           ON b.player_id = au.player_id
          AND b.tenant_plat_id = :tenant_plat_id
          AND b.channel_id = :channel_id
          AND b.settle_status = 1
          AND b.settle_time >= :start_date
          AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY au.player_id
),
ranked_users AS (
    SELECT
        br.player_id,
        br.total_valid_bet_amount,
        ROW_NUMBER() OVER (ORDER BY br.total_valid_bet_amount DESC, br.player_id) AS bet_rank
    FROM bet_rank_base br
),
segment_def AS (
    SELECT '全部用户' AS user_segment, 1 AS segment_sort
    UNION ALL SELECT CONCAT('TOP', :top_n), 2
    UNION ALL SELECT CONCAT('非TOP', :top_n), 3
    UNION ALL SELECT 'VIP0', 10
    UNION ALL SELECT 'VIP1', 11
    UNION ALL SELECT 'VIP2', 12
    UNION ALL SELECT 'VIP3', 13
),
user_segments AS (
    SELECT '全部用户' AS user_segment, 1 AS segment_sort, aud.player_id
    FROM active_user_dim aud
    UNION ALL
    SELECT CASE WHEN ru.bet_rank <= :top_n THEN CONCAT('TOP', :top_n) ELSE CONCAT('非TOP', :top_n) END AS user_segment,
           CASE WHEN ru.bet_rank <= :top_n THEN 2 ELSE 3 END AS segment_sort,
           ru.player_id
    FROM ranked_users ru
    UNION ALL
    SELECT CONCAT('VIP', aud.vip_id) AS user_segment,
           10 + aud.vip_id AS segment_sort,
           aud.player_id
    FROM active_user_dim aud
    WHERE aud.vip_id BETWEEN 0 AND 3
),
segment_user_count AS (
    SELECT
        user_segment,
        segment_sort,
        COUNT(DISTINCT player_id) AS user_count
    FROM user_segments
    GROUP BY user_segment, segment_sort
),
deposit_summary AS (
    SELECT
        us.user_segment,
        us.segment_sort,
        COUNT(DISTINCT d.player_id) AS deposit_user_count,
        SUM(d.actual_amount) AS deposit_amount
    FROM user_segments us
    LEFT JOIN dwd_order_deposit d
           ON d.player_id = us.player_id
          AND d.status = 2
          AND d.tenant_plat_id = :tenant_plat_id
          AND d.channel_id = :channel_id
          AND d.callback_time >= :start_date
          AND d.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY us.user_segment, us.segment_sort
),
withdraw_summary AS (
    SELECT
        us.user_segment,
        us.segment_sort,
        COUNT(DISTINCT w.player_id) AS withdrawal_user_count,
        SUM(w.act_amount) AS withdrawal_amount
    FROM user_segments us
    LEFT JOIN dwd_order_withdrawal w
           ON w.player_id = us.player_id
          AND w.status = 3
          AND w.tenant_plat_id = :tenant_plat_id
          AND w.channel_id = :channel_id
          AND w.callback_time >= :start_date
          AND w.callback_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY us.user_segment, us.segment_sort
),
bet_summary AS (
    SELECT
        us.user_segment,
        us.segment_sort,
        COUNT(DISTINCT b.player_id) AS bet_user_count,
        SUM(b.valid_bet_amount) AS valid_bet_amount,
        SUM(b.win_loss_amount) AS win_loss_amount
    FROM user_segments us
    LEFT JOIN dwd_bet_order b
           ON b.player_id = us.player_id
          AND b.settle_status = 1
          AND b.tenant_plat_id = :tenant_plat_id
          AND b.channel_id = :channel_id
          AND b.settle_time >= :start_date
          AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY us.user_segment, us.segment_sort
)
SELECT
    sd.user_segment AS `用户分层`,
    COALESCE(suc.user_count, 0) AS `用户数`,
    COALESCE(d.deposit_user_count, 0) AS `存款人数`,
    COALESCE(d.deposit_amount, 0) AS `存款金额`,
    COALESCE(w.withdrawal_user_count, 0) AS `提现人数`,
    COALESCE(w.withdrawal_amount, 0) AS `提现金额`,
    COALESCE(d.deposit_amount, 0) - COALESCE(w.withdrawal_amount, 0) AS `充提差`,
    COALESCE(b.bet_user_count, 0) AS `投注人数`,
    COALESCE(b.valid_bet_amount, 0) AS `有效投注`,
    COALESCE(b.win_loss_amount, 0) AS `输赢`,
    ROUND(COALESCE(b.win_loss_amount, 0) / NULLIF(COALESCE(b.valid_bet_amount, 0), 0), 6) AS `杀率`,
    ROUND(COALESCE(b.valid_bet_amount, 0) / NULLIF(COALESCE(d.deposit_amount, 0), 0), 6) AS `投充比`
FROM segment_def sd
LEFT JOIN segment_user_count suc
       ON suc.user_segment = sd.user_segment
      AND suc.segment_sort = sd.segment_sort
LEFT JOIN deposit_summary d
       ON d.user_segment = sd.user_segment
      AND d.segment_sort = sd.segment_sort
LEFT JOIN withdraw_summary w
       ON w.user_segment = sd.user_segment
      AND w.segment_sort = sd.segment_sort
LEFT JOIN bet_summary b
       ON b.user_segment = sd.user_segment
      AND b.segment_sort = sd.segment_sort
ORDER BY sd.segment_sort;
```

## 备注

- 当前模板直接输出全部用户、TOPN、非TOPN、VIP0~VIP3固定分层；VIP分层来自 `dim_player.vip_id`。
- “大户”“头部用户”“投注流水最高的前 N 个用户”默认等价于按区间累计有效投注额排序的 `TOPN` 用户。
- 当活跃用户不足 `:top_n` 人时，`TOPN` 返回实际活跃用户，`NON_TOPN` 返回 0 行 / 0 值；回答中应说明该边界，不要继续追问排序依据。
- SQL 按 TiDB / MySQL 8 风格编写；导入前需在实际 runtime datasource 下做一次校验。
- 当前可视为 SQL 草案，校验通过后可转为 sql_pair。
