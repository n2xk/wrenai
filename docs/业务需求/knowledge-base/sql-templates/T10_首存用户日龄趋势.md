# T10 首存用户日龄趋势

## 模板用途

统计首存 cohort 从首日开始的 D1~DN 投充比/杀率趋势。

## 建议问题（可转为 sql_pair.question）

- 统计首存 cohort 从首日开始的 D1~DN 投充比/杀率趋势
- 统计某渠道首存 cohort 在首存后 N 天内的日龄趋势指标

## 核心表/模型

- dwd_order_deposit
- dwd_order_withdrawal
- dwd_bet_order

## 参数

- tenant_plat_id
- channel_id
- cohort_start_date
- cohort_end_date
- n_days

## SQL 模板

```sql
WITH RECURSIVE seq AS (
    SELECT 1 AS relative_day_no
    UNION ALL
    SELECT relative_day_no + 1
    FROM seq
    WHERE relative_day_no < :n_days
),
top_limit AS (
    SELECT COALESCE(:top_n, 3) AS top_n
),
first_deposit_cohort AS (
    SELECT
        d.tenant_plat_id,
        d.channel_id,
        d.player_id,
        DATE(MIN(d.callback_time)) AS first_deposit_date
    FROM dwd_order_deposit d
    WHERE d.status = 2
      AND d.times = 1
      AND d.tenant_plat_id = :tenant_plat_id
      AND d.channel_id = :channel_id
      AND d.callback_time >= :cohort_start_date
      AND d.callback_time < DATE_ADD(:cohort_end_date, INTERVAL 1 DAY)
    GROUP BY d.tenant_plat_id, d.channel_id, d.player_id
),
rank_base AS (
    SELECT
        c.tenant_plat_id,
        c.channel_id,
        c.player_id,
        COALESCE(SUM(b.valid_bet_amount), 0) AS total_valid_bet_amount
    FROM first_deposit_cohort c
    LEFT JOIN dwd_bet_order b
           ON b.player_id = c.player_id
          AND b.tenant_plat_id = c.tenant_plat_id
          AND b.channel_id = c.channel_id
          AND b.settle_status = 1
          AND b.settle_time >= c.first_deposit_date
          AND b.settle_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.tenant_plat_id, c.channel_id, c.player_id
),
ranked_cohort AS (
    SELECT
        c.*,
        ROW_NUMBER() OVER (
            ORDER BY rb.total_valid_bet_amount DESC, c.player_id
        ) AS bet_rank
    FROM first_deposit_cohort c
    INNER JOIN rank_base rb
            ON rb.tenant_plat_id = c.tenant_plat_id
           AND rb.channel_id = c.channel_id
           AND rb.player_id = c.player_id
),
cohort_segments AS (
    SELECT
        tenant_plat_id,
        channel_id,
        player_id,
        first_deposit_date,
        '全部' AS user_segment
    FROM ranked_cohort
    UNION ALL
    SELECT
        rc.tenant_plat_id,
        rc.channel_id,
        rc.player_id,
        rc.first_deposit_date,
        CASE
            WHEN rc.bet_rank <= tl.top_n THEN CONCAT('TOP', tl.top_n)
            ELSE CONCAT('非TOP', tl.top_n)
        END AS user_segment
    FROM ranked_cohort rc
    CROSS JOIN top_limit tl
),
cohort_size AS (
    SELECT
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        user_segment,
        COUNT(DISTINCT player_id) AS cohort_user_count
    FROM cohort_segments
    GROUP BY tenant_plat_id, channel_id, first_deposit_date, user_segment
),
deposit_daily AS (
    SELECT
        c.player_id,
        DATE(d.callback_time) AS event_date,
        SUM(d.actual_amount) AS deposit_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_order_deposit d
            ON d.player_id = c.player_id
           AND d.tenant_plat_id = c.tenant_plat_id
           AND d.channel_id = c.channel_id
    WHERE d.status = 2
      AND d.callback_time >= c.first_deposit_date
      AND d.callback_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(d.callback_time)
),
withdraw_daily AS (
    SELECT
        c.player_id,
        DATE(w.callback_time) AS event_date,
        SUM(w.act_amount) AS withdrawal_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_order_withdrawal w
            ON w.player_id = c.player_id
           AND w.tenant_plat_id = c.tenant_plat_id
           AND w.channel_id = c.channel_id
    WHERE w.status = 3
      AND w.callback_time >= c.first_deposit_date
      AND w.callback_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(w.callback_time)
),
bet_daily AS (
    SELECT
        c.player_id,
        DATE(b.settle_time) AS event_date,
        SUM(b.valid_bet_amount) AS valid_bet_amount,
        SUM(b.win_loss_amount) AS win_loss_amount
    FROM first_deposit_cohort c
    INNER JOIN dwd_bet_order b
            ON b.player_id = c.player_id
           AND b.tenant_plat_id = c.tenant_plat_id
           AND b.channel_id = c.channel_id
    WHERE b.settle_status = 1
      AND b.settle_time >= c.first_deposit_date
      AND b.settle_time < DATE_ADD(c.first_deposit_date, INTERVAL :n_days DAY)
    GROUP BY c.player_id, DATE(b.settle_time)
),
player_day_keys AS (
    SELECT player_id, event_date FROM deposit_daily
    UNION
    SELECT player_id, event_date FROM withdraw_daily
    UNION
    SELECT player_id, event_date FROM bet_daily
),
player_day_metrics AS (
    SELECT
        c.tenant_plat_id,
        c.channel_id,
        c.first_deposit_date,
        DATEDIFF(k.event_date, c.first_deposit_date) + 1 AS relative_day_no,
        k.player_id,
        d.deposit_amount,
        w.withdrawal_amount,
        b.valid_bet_amount,
        b.win_loss_amount
    FROM first_deposit_cohort c
    INNER JOIN player_day_keys k
            ON k.player_id = c.player_id
    LEFT JOIN deposit_daily d
           ON d.player_id = k.player_id
          AND d.event_date = k.event_date
    LEFT JOIN withdraw_daily w
           ON w.player_id = k.player_id
          AND w.event_date = k.event_date
    LEFT JOIN bet_daily b
           ON b.player_id = k.player_id
          AND b.event_date = k.event_date
    WHERE DATEDIFF(k.event_date, c.first_deposit_date) + 1 BETWEEN 1 AND :n_days
),
daily_agg AS (
    SELECT
        s.tenant_plat_id,
        s.channel_id,
        s.first_deposit_date,
        s.user_segment,
        p.relative_day_no,
        SUM(CASE WHEN p.deposit_amount IS NOT NULL THEN 1 ELSE 0 END) AS deposit_user_count,
        SUM(COALESCE(p.deposit_amount, 0)) AS deposit_amount,
        SUM(CASE WHEN p.withdrawal_amount IS NOT NULL THEN 1 ELSE 0 END) AS withdrawal_user_count,
        SUM(COALESCE(p.withdrawal_amount, 0)) AS withdrawal_amount,
        SUM(CASE WHEN p.valid_bet_amount IS NOT NULL THEN 1 ELSE 0 END) AS bet_user_count,
        SUM(COALESCE(p.valid_bet_amount, 0)) AS valid_bet_amount,
        SUM(COALESCE(p.win_loss_amount, 0)) AS win_loss_amount
    FROM cohort_segments s
    LEFT JOIN player_day_metrics p
           ON p.tenant_plat_id = s.tenant_plat_id
          AND p.channel_id = s.channel_id
          AND p.first_deposit_date = s.first_deposit_date
          AND p.player_id = s.player_id
    GROUP BY s.tenant_plat_id, s.channel_id, s.first_deposit_date, s.user_segment, p.relative_day_no
),
day_metrics AS (
    SELECT
        cs.tenant_plat_id,
        cs.channel_id,
        cs.first_deposit_date,
        cs.user_segment,
        cs.cohort_user_count,
        seq.relative_day_no,
        COALESCE(a.deposit_user_count, 0) AS deposit_user_count,
        COALESCE(a.deposit_amount, 0) AS deposit_amount,
        COALESCE(a.withdrawal_user_count, 0) AS withdrawal_user_count,
        COALESCE(a.withdrawal_amount, 0) AS withdrawal_amount,
        COALESCE(a.deposit_amount, 0) - COALESCE(a.withdrawal_amount, 0) AS charge_withdraw_diff,
        COALESCE(a.bet_user_count, 0) AS bet_user_count,
        COALESCE(a.valid_bet_amount, 0) AS valid_bet_amount,
        COALESCE(a.win_loss_amount, 0) AS win_loss_amount,
        ROUND(COALESCE(a.win_loss_amount, 0) / NULLIF(COALESCE(a.valid_bet_amount, 0), 0), 6) AS kill_rate,
        ROUND(COALESCE(a.valid_bet_amount, 0) / NULLIF(COALESCE(a.deposit_amount, 0), 0), 6) AS bet_deposit_ratio
    FROM cohort_size cs
    CROSS JOIN seq
    LEFT JOIN daily_agg a
           ON a.tenant_plat_id = cs.tenant_plat_id
          AND a.channel_id = cs.channel_id
          AND a.first_deposit_date = cs.first_deposit_date
          AND a.user_segment = cs.user_segment
          AND a.relative_day_no = seq.relative_day_no
),
pivot_result AS (
    SELECT
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        user_segment,
        cohort_user_count,
        MAX(CASE WHEN relative_day_no = 1 THEN kill_rate END) AS kill_rate_1_day,
        MAX(CASE WHEN relative_day_no = 2 THEN kill_rate END) AS kill_rate_2_day,
        MAX(CASE WHEN relative_day_no = 3 THEN kill_rate END) AS kill_rate_3_day,
        MAX(CASE WHEN relative_day_no = 4 THEN kill_rate END) AS kill_rate_4_day,
        MAX(CASE WHEN relative_day_no = 5 THEN kill_rate END) AS kill_rate_5_day,
        MAX(CASE WHEN relative_day_no = 6 THEN kill_rate END) AS kill_rate_6_day,
        MAX(CASE WHEN relative_day_no = 7 THEN kill_rate END) AS kill_rate_7_day,
        MAX(CASE WHEN relative_day_no = 15 THEN kill_rate END) AS kill_rate_15_day,
        MAX(CASE WHEN relative_day_no = 30 THEN kill_rate END) AS kill_rate_30_day,
        MAX(CASE WHEN relative_day_no = 1 THEN bet_deposit_ratio END) AS bet_deposit_ratio_1_day,
        MAX(CASE WHEN relative_day_no = 2 THEN bet_deposit_ratio END) AS bet_deposit_ratio_2_day,
        MAX(CASE WHEN relative_day_no = 3 THEN bet_deposit_ratio END) AS bet_deposit_ratio_3_day,
        MAX(CASE WHEN relative_day_no = 4 THEN bet_deposit_ratio END) AS bet_deposit_ratio_4_day,
        MAX(CASE WHEN relative_day_no = 5 THEN bet_deposit_ratio END) AS bet_deposit_ratio_5_day,
        MAX(CASE WHEN relative_day_no = 6 THEN bet_deposit_ratio END) AS bet_deposit_ratio_6_day,
        MAX(CASE WHEN relative_day_no = 7 THEN bet_deposit_ratio END) AS bet_deposit_ratio_7_day,
        MAX(CASE WHEN relative_day_no = 15 THEN bet_deposit_ratio END) AS bet_deposit_ratio_15_day,
        MAX(CASE WHEN relative_day_no = 30 THEN bet_deposit_ratio END) AS bet_deposit_ratio_30_day
    FROM day_metrics
    GROUP BY tenant_plat_id, channel_id, first_deposit_date, user_segment, cohort_user_count
),
summary_size AS (
    SELECT
        tenant_plat_id,
        channel_id,
        user_segment,
        SUM(cohort_user_count) AS cohort_user_count
    FROM cohort_size
    GROUP BY tenant_plat_id, channel_id, user_segment
),
summary_day_metrics AS (
    SELECT
        tenant_plat_id,
        channel_id,
        user_segment,
        relative_day_no,
        ROUND(SUM(win_loss_amount) / NULLIF(SUM(valid_bet_amount), 0), 6) AS kill_rate,
        ROUND(SUM(valid_bet_amount) / NULLIF(SUM(deposit_amount), 0), 6) AS bet_deposit_ratio
    FROM day_metrics
    GROUP BY tenant_plat_id, channel_id, user_segment, relative_day_no
),
summary_pivot AS (
    SELECT
        s.tenant_plat_id,
        s.channel_id,
        NULL AS first_deposit_date,
        s.user_segment,
        s.cohort_user_count,
        MAX(CASE WHEN m.relative_day_no = 1 THEN m.kill_rate END) AS kill_rate_1_day,
        MAX(CASE WHEN m.relative_day_no = 2 THEN m.kill_rate END) AS kill_rate_2_day,
        MAX(CASE WHEN m.relative_day_no = 3 THEN m.kill_rate END) AS kill_rate_3_day,
        MAX(CASE WHEN m.relative_day_no = 4 THEN m.kill_rate END) AS kill_rate_4_day,
        MAX(CASE WHEN m.relative_day_no = 5 THEN m.kill_rate END) AS kill_rate_5_day,
        MAX(CASE WHEN m.relative_day_no = 6 THEN m.kill_rate END) AS kill_rate_6_day,
        MAX(CASE WHEN m.relative_day_no = 7 THEN m.kill_rate END) AS kill_rate_7_day,
        MAX(CASE WHEN m.relative_day_no = 15 THEN m.kill_rate END) AS kill_rate_15_day,
        MAX(CASE WHEN m.relative_day_no = 30 THEN m.kill_rate END) AS kill_rate_30_day,
        MAX(CASE WHEN m.relative_day_no = 1 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_1_day,
        MAX(CASE WHEN m.relative_day_no = 2 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_2_day,
        MAX(CASE WHEN m.relative_day_no = 3 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_3_day,
        MAX(CASE WHEN m.relative_day_no = 4 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_4_day,
        MAX(CASE WHEN m.relative_day_no = 5 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_5_day,
        MAX(CASE WHEN m.relative_day_no = 6 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_6_day,
        MAX(CASE WHEN m.relative_day_no = 7 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_7_day,
        MAX(CASE WHEN m.relative_day_no = 15 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_15_day,
        MAX(CASE WHEN m.relative_day_no = 30 THEN m.bet_deposit_ratio END) AS bet_deposit_ratio_30_day
    FROM summary_size s
    LEFT JOIN summary_day_metrics m
           ON m.tenant_plat_id = s.tenant_plat_id
          AND m.channel_id = s.channel_id
          AND m.user_segment = s.user_segment
    GROUP BY s.tenant_plat_id, s.channel_id, s.user_segment, s.cohort_user_count
),
combined_result AS (
    SELECT
        '汇总' AS row_type,
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        user_segment,
        cohort_user_count,
        kill_rate_1_day,
        kill_rate_2_day,
        kill_rate_3_day,
        kill_rate_4_day,
        kill_rate_5_day,
        kill_rate_6_day,
        kill_rate_7_day,
        kill_rate_15_day,
        kill_rate_30_day,
        bet_deposit_ratio_1_day,
        bet_deposit_ratio_2_day,
        bet_deposit_ratio_3_day,
        bet_deposit_ratio_4_day,
        bet_deposit_ratio_5_day,
        bet_deposit_ratio_6_day,
        bet_deposit_ratio_7_day,
        bet_deposit_ratio_15_day,
        bet_deposit_ratio_30_day
    FROM summary_pivot
    UNION ALL
    SELECT
        '明细' AS row_type,
        tenant_plat_id,
        channel_id,
        first_deposit_date,
        user_segment,
        cohort_user_count,
        kill_rate_1_day,
        kill_rate_2_day,
        kill_rate_3_day,
        kill_rate_4_day,
        kill_rate_5_day,
        kill_rate_6_day,
        kill_rate_7_day,
        kill_rate_15_day,
        kill_rate_30_day,
        bet_deposit_ratio_1_day,
        bet_deposit_ratio_2_day,
        bet_deposit_ratio_3_day,
        bet_deposit_ratio_4_day,
        bet_deposit_ratio_5_day,
        bet_deposit_ratio_6_day,
        bet_deposit_ratio_7_day,
        bet_deposit_ratio_15_day,
        bet_deposit_ratio_30_day
    FROM pivot_result
),
with_dims AS (
    SELECT
        cr.*,
        tp.name AS site_name,
        ch.channel_partner_username AS channel_partner_name,
        ch.name AS channel_name
    FROM combined_result cr
    LEFT JOIN tenant_plat tp
           ON tp.id = cr.tenant_plat_id
    LEFT JOIN channel ch
           ON ch.id = cr.channel_id
          AND ch.tenant_plat_id = cr.tenant_plat_id
)
SELECT
    CASE
        WHEN row_type = '汇总' THEN '汇总'
        ELSE CAST(first_deposit_date AS CHAR)
    END AS `日期`,
    site_name AS `站点名称`,
    channel_partner_name AS `所属渠道商`,
    channel_name AS `渠道名称`,
    '首存用户' AS `用户属性`,
    user_segment AS `用户类型`,
    cohort_user_count AS `用户人数`,
    kill_rate_1_day AS `首日杀率`,
    kill_rate_2_day AS `2日杀率`,
    kill_rate_3_day AS `3日杀率`,
    kill_rate_4_day AS `4日杀率`,
    kill_rate_5_day AS `5日杀率`,
    kill_rate_6_day AS `6日杀率`,
    kill_rate_7_day AS `7日杀率`,
    kill_rate_15_day AS `15日杀率`,
    kill_rate_30_day AS `30日杀率`,
    bet_deposit_ratio_1_day AS `首日投充比`,
    bet_deposit_ratio_2_day AS `2日投充比`,
    bet_deposit_ratio_3_day AS `3日投充比`,
    bet_deposit_ratio_4_day AS `4日投充比`,
    bet_deposit_ratio_5_day AS `5日投充比`,
    bet_deposit_ratio_6_day AS `6日投充比`,
    bet_deposit_ratio_7_day AS `7日投充比`,
    bet_deposit_ratio_15_day AS `15日投充比`,
    bet_deposit_ratio_30_day AS `30日投充比`
FROM with_dims
ORDER BY
    CASE row_type WHEN '汇总' THEN 0 ELSE 1 END,
    first_deposit_date,
    CASE user_segment WHEN '全部' THEN 0 WHEN CONCAT('TOP', COALESCE(:top_n, 3)) THEN 1 ELSE 2 END;
```

## 备注

- `D1` 表示首存当日，`D2` 表示首存次日，以此类推。
- 统计对象仅限 `cohort_start_date ~ cohort_end_date` 内的首存用户，但这些用户的后续行为可延伸到 cohort 窗口之外。
- 若运行环境不支持递归 CTE，可改成数字维表/日期维表实现。
- SQL 按 TiDB / MySQL 8 风格编写；导入前需在实际 runtime datasource 下做一次校验。
- 当前可视为 SQL 草案，校验通过后可转为 sql_pair。
