# T12 TOP3/5 游戏类型分层

## 模板用途

针对所有用户、TOP3/5 用户与非 TOP3/5 用户输出游戏类型流水分布透视宽表。

## 建议问题（可转为 sql_pair.question）

- 生成所有用户、TOP3、非TOP3按游戏类型为列的流水分布透视宽表
- 对比某渠道 TOPN 与非TOPN 用户在各游戏类型上的投注分布

## 核心表/模型

- dwd_bet_order
- bds_bet_order_detail
- dim_game_type
- dim_game_line
- dim_game_line_series

## 参数

- tenant_plat_id
- channel_id
- start_date
- end_date
- top_n

## SQL 模板

```sql
WITH bet_rank_base AS (
    SELECT
        b.player_id,
        SUM(b.valid_bet_amount) AS total_valid_bet_amount
    FROM dwd_bet_order b
    WHERE b.settle_status = 1
      AND b.tenant_plat_id = :tenant_plat_id
      AND b.channel_id = :channel_id
      AND b.settle_time >= :start_date
      AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
    GROUP BY b.player_id
),
ranked_users AS (
    SELECT
        br.player_id,
        ROW_NUMBER() OVER (
            ORDER BY br.total_valid_bet_amount DESC, br.player_id
        ) AS bet_rank
    FROM bet_rank_base br
),
base_bets AS (
    SELECT
        b.player_id,
        CASE
            WHEN gt.name = '体育' THEN '体育'
            WHEN gt.name = '棋牌' THEN '电子棋牌'
            WHEN gt.name = '电子' THEN '电子-老虎机'
            WHEN gt.name IN ('捕鱼','彩票','电竞','电子-街机','视讯','电子-奖池','真人棋牌','棋牌锦标赛') THEN gt.name
            ELSE COALESCE(gt.name, '其他游戏类型')
        END AS dim_game_type_bucket,
        b.valid_bet_amount,
        b.win_loss_amount
    FROM bds_bet_order_detail b
    LEFT JOIN dim_game_type gt
           ON gt.id = b.game_type_id
    WHERE b.settle_status = 1
      AND b.game_type_id IS NOT NULL
      AND b.tenant_plat_id = :tenant_plat_id
      AND b.channel_id = :channel_id
      AND b.settle_time >= :start_date
      AND b.settle_time < DATE_ADD(:end_date, INTERVAL 1 DAY)
),
segmented_bets AS (
    SELECT
        '所有用户' AS user_segment,
        1 AS segment_sort,
        bb.dim_game_type_bucket,
        bb.valid_bet_amount,
        bb.win_loss_amount
    FROM base_bets bb
    UNION ALL
    SELECT
        CASE WHEN COALESCE(ru.bet_rank, 999999) <= :top_n THEN CONCAT('TOP', :top_n) ELSE CONCAT('非TOP', :top_n) END AS user_segment,
        CASE WHEN COALESCE(ru.bet_rank, 999999) <= :top_n THEN 2 ELSE 3 END AS segment_sort,
        bb.dim_game_type_bucket,
        bb.valid_bet_amount,
        bb.win_loss_amount
    FROM base_bets bb
    LEFT JOIN ranked_users ru
           ON ru.player_id = bb.player_id
),
agg AS (
    SELECT
        user_segment,
        segment_sort,
        dim_game_type_bucket,
        COUNT(*) AS bet_times,
        SUM(valid_bet_amount) AS valid_bet_amount,
        SUM(win_loss_amount) AS win_loss_amount
    FROM segmented_bets
    GROUP BY user_segment, segment_sort, dim_game_type_bucket
),
segment_total AS (
    SELECT
        user_segment,
        segment_sort,
        SUM(bet_times) AS total_bet_times,
        SUM(valid_bet_amount) AS total_valid_bet_amount,
        SUM(win_loss_amount) AS total_win_loss_amount
    FROM agg
    GROUP BY user_segment, segment_sort
),
metric_source AS (
    SELECT
        a.user_segment,
        a.segment_sort,
        a.dim_game_type_bucket,
        '有效投注流水' AS metric_name,
        a.valid_bet_amount AS metric_value,
        st.total_valid_bet_amount AS total_value
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
    UNION ALL
    SELECT a.user_segment, a.segment_sort, a.dim_game_type_bucket, '下注次数', a.bet_times, st.total_bet_times
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
    UNION ALL
    SELECT a.user_segment, a.segment_sort, a.dim_game_type_bucket, '均注金额', ROUND(a.valid_bet_amount / NULLIF(a.bet_times, 0), 4), ROUND(st.total_valid_bet_amount / NULLIF(st.total_bet_times, 0), 4)
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
    UNION ALL
    SELECT a.user_segment, a.segment_sort, a.dim_game_type_bucket, '输赢', a.win_loss_amount, st.total_win_loss_amount
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
    UNION ALL
    SELECT a.user_segment, a.segment_sort, a.dim_game_type_bucket, '杀率', ROUND(a.win_loss_amount / NULLIF(a.valid_bet_amount, 0), 6), ROUND(st.total_win_loss_amount / NULLIF(st.total_valid_bet_amount, 0), 6)
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
    UNION ALL
    SELECT a.user_segment, a.segment_sort, a.dim_game_type_bucket, '投注占比', ROUND(a.valid_bet_amount / NULLIF(st.total_valid_bet_amount, 0), 6), 1
    FROM agg a
    INNER JOIN segment_total st ON st.user_segment = a.user_segment AND st.segment_sort = a.segment_sort
)
SELECT
    ms.user_segment AS `用户分层`,
    ms.metric_name AS `指标`,
    ROUND(MAX(ms.total_value), 6) AS `合计`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '捕鱼' THEN ms.metric_value ELSE 0 END), 6) AS `捕鱼`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '彩票' THEN ms.metric_value ELSE 0 END), 6) AS `彩票`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '电竞' THEN ms.metric_value ELSE 0 END), 6) AS `电竞`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '电子-老虎机' THEN ms.metric_value ELSE 0 END), 6) AS `电子-老虎机`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '电子-街机' THEN ms.metric_value ELSE 0 END), 6) AS `电子-街机`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '电子棋牌' THEN ms.metric_value ELSE 0 END), 6) AS `电子棋牌`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '体育' THEN ms.metric_value ELSE 0 END), 6) AS `体育`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '视讯' THEN ms.metric_value ELSE 0 END), 6) AS `视讯`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '电子-奖池' THEN ms.metric_value ELSE 0 END), 6) AS `电子-奖池`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '真人棋牌' THEN ms.metric_value ELSE 0 END), 6) AS `真人棋牌`,
    ROUND(SUM(CASE WHEN ms.dim_game_type_bucket = '棋牌锦标赛' THEN ms.metric_value ELSE 0 END), 6) AS `棋牌锦标赛`
FROM metric_source ms
GROUP BY ms.user_segment, ms.segment_sort, ms.metric_name
ORDER BY ms.segment_sort,
         FIELD(ms.metric_name, '有效投注流水', '下注次数', '均注金额', '输赢', '杀率', '投注占比');
```
